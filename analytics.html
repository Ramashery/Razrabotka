      Установка трекера
      Трекер уже подключён через /dc-tracker.js — данные пишутся в Firestore от всех посетителей
      Коллекции Firestore:
      • analytics_sessions — одна запись на сессию (устройство, браузер, ОС, источник, isNew)
      • analytics_pageviews — одна запись на просмотр страницы
      • analytics_events — кастомные события через dcTrack('name', props)
      Кастомные события — вызывайте в любом месте сайта:
      // Клик по кнопке контакта
dcTrack('click', { label: 'contact_button' });

// Просмотр проекта
dcTrack('view_project', { slug: 'my-project' });

// Форма отправлена
dcTrack('form_submit', { form: 'contact' });
      Firestore Security Rules — добавьте в консоли Firebase:
      // Firestore Rules — разрешить только запись в аналитические коллекции
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Аналитика: только добавление новых документов (не чтение/удаление)
    match /analytics_sessions/{doc} {
      allow create: if true;
      allow read, update, delete: if false;
    }
    match /analytics_pageviews/{doc} {
      allow create: if true;
      allow read, update, delete: if false;
    }
    match /analytics_events/{doc} {
      allow create: if true;
      allow read, update, delete: if false;
    }
    // Остальные коллекции — ваши текущие правила
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
